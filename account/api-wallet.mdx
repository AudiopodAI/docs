## API Reference

### Get Balance

Check your current wallet balance.

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X GET "https://api.audiopod.ai/api/v1/api-wallet/balance" \
      -H "X-API-Key: $AUDIOPOD_API_KEY"
    ```
  </Tab>
  <Tab title="Python">
    ```python
    import requests

    API_KEY = "your_api_key_here"

    response = requests.get(
        "https://api.audiopod.ai/api/v1/api-wallet/balance",
        headers={"X-API-Key": API_KEY}
    )

    balance = response.json()
    print(f"Balance: {balance['balance_usd']}")
    print(f"Total spent: {balance['total_spent_usd']}")
    ```
  </Tab>
  <Tab title="Node.js">
    ```javascript
    const axios = require('axios');

    const API_KEY = 'your_api_key_here';

    async function getBalance() {
      const { data } = await axios.get(
        'https://api.audiopod.ai/api/v1/api-wallet/balance',
        { headers: { 'X-API-Key': API_KEY } }
      );

      console.log(`Balance: ${data.balance_usd}`);
      console.log(`Total spent: ${data.total_spent_usd}`);
    }

    getBalance();
    ```
  </Tab>
</Tabs>

**Response:**

```json
{
  "balance_cents": 2500,
  "balance_usd": "$25.00",
  "total_topup_cents": 5000,
  "total_topup_usd": "$50.00",
  "total_spent_cents": 2500,
  "total_spent_usd": "$25.00",
  "low_balance_warning": false,
  "warning_threshold_cents": 500
}
```

### Get Pricing

Get pricing table for all services. **No authentication required.**

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X GET "https://api.audiopod.ai/api/v1/api-wallet/pricing"
    ```
  </Tab>
  <Tab title="Python">
    ```python
    import requests

    response = requests.get("https://api.audiopod.ai/api/v1/api-wallet/pricing")
    pricing = response.json()

    print("Service Pricing:")
    for service, info in pricing["services"].items():
        print(f"  {service}: {info['rate_per_minute']}")
    ```
  </Tab>
  <Tab title="Node.js">
    ```javascript
    const axios = require('axios');
    ```

### Estimate Cost

Calculate cost before processing. **No authentication required.**

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X POST "https://api.audiopod.ai/api/v1/api-wallet/estimate" \
      -H "Content-Type: application/json" \
      -d '{"service_type": "stem_extraction", "duration_seconds": 300}'
    ```
  </Tab>
  <Tab title="Python">
    ```python
    import requests

    response = requests.post(
        "https://api.audiopod.ai/api/v1/api-wallet/estimate",
        json={
            "service_type": "stem_extraction",
            "duration_seconds": 300  # 5 minutes
        }
    )

    estimate = response.json()
    print(f"Duration: {estimate['duration_minutes']} minutes")
    print(f"Estimated cost: {estimate['cost_usd']}")
    ```
  </Tab>
  <Tab title="Node.js">
    ```javascript
    const axios = require('axios');

    async function estimateCost() {
      const { data } = await axios.post(
        'https://api.audiopod.ai/api/v1/api-wallet/estimate',
        {
          service_type: 'stem_extraction',
          duration_seconds: 300  // 5 minutes
        }
      );

      console.log(`Duration: ${data.duration_minutes} minutes`);
      console.log(`Estimated cost: ${data.cost_usd}`);
    }

    estimateCost();
    ```
  </Tab>
</Tabs>

**Response:**

```json
{
  "service_type": "stem_extraction",
  "duration_seconds": 300,

  "cost_cents": 50,
  "cost_usd": "$0.50",
```

### Top Up Wallet

Create a Stripe Checkout session to add funds.

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X POST "https://api.audiopod.ai/api/v1/api-wallet/topup/checkout" \
      -H "X-API-Key: $AUDIOPOD_API_KEY" \
      -H "Content-Type: application/json" \
      -d '{"amount_cents": 2500}'
    ```
  </Tab>
  <Tab title="Python">
    ```python
    import requests

    API_KEY = "your_api_key_here"

    response = requests.post(
        "https://api.audiopod.ai/api/v1/api-wallet/topup/checkout",
        headers={"X-API-Key": API_KEY},
        json={"amount_cents": 2500}  # $25.00
    )

    checkout = response.json()
    print(f"Complete payment at: {checkout['url']}")
    ```
  </Tab>
  <Tab title="Node.js">
    ```javascript
    const axios = require('axios');

    const API_KEY = 'your_api_key_here';

    async function topUp() {
      const { data } = await axios.post(
        'https://api.audiopod.ai/api/v1/api-wallet/topup/checkout',
        { amount_cents: 2500 },  // $25.00
        { headers: { 'X-API-Key': API_KEY } }
      );

      console.log(`Complete payment at: ${data.url}`);
    }

    topUp();
    ```
  </Tab>
</Tabs>

**Response:**

```json

  "url": "https://checkout.stripe.com/c/pay/cs_live_abc123..."
}
```

### Get Usage History

View your API usage logs.

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X GET "https://api.audiopod.ai/api/v1/api-wallet/usage?page=1&limit=50" \
      -H "X-API-Key: $AUDIOPOD_API_KEY"
    ```
  </Tab>
  <Tab title="Python">
    ```python
    import requests

    API_KEY = "your_api_key_here"

    response = requests.get(
        "https://api.audiopod.ai/api/v1/api-wallet/usage",
        headers={"X-API-Key": API_KEY},
        params={"page": 1, "limit": 50}
    )

    usage = response.json()
    print(f"Recent usage:")
    for log in usage["logs"]:
        print(f"  {log['service_type']}: {log['amount_usd']} ({log['duration_minutes']} min)")
    ```
  </Tab>
  <Tab title="Node.js">
    ```javascript
    const axios = require('axios');

    const API_KEY = 'your_api_key_here';

    async function getUsage() {
      const { data } = await axios.get(
        'https://api.audiopod.ai/api/v1/api-wallet/usage',
        {
          headers: { 'X-API-Key': API_KEY },
          params: { page: 1, limit: 50 }
        }
      );

      console.log('Recent usage:');
      data.logs.forEach(log => {
        console.log(`  ${log.service_type}: ${log.amount_usd} (${log.duration_minutes} min)`);
      });
    }

    getUsage();
    ```

### Get Top-up History

View your wallet top-up transactions.

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X GET "https://api.audiopod.ai/api/v1/api-wallet/topups?page=1&limit=20" \
      -H "X-API-Key: $AUDIOPOD_API_KEY"
    ```
  </Tab>
  <Tab title="Python">
    ```python
    import requests

    API_KEY = "your_api_key_here"

    response = requests.get(
        "https://api.audiopod.ai/api/v1/api-wallet/topups",
        headers={"X-API-Key": API_KEY},
        params={"page": 1, "limit": 20}
    )

    topups = response.json()
    print("Top-up history:")
    for topup in topups["topups"]:
        print(f"  {topup['created_at']}: {topup['amount_usd']} ({topup['status']})")
    ```
  </Tab>
  <Tab title="Node.js">
    ```javascript
    const axios = require('axios');

    const API_KEY = 'your_api_key_here';

    async function getTopups() {

        'https://api.audiopod.ai/api/v1/api-wallet/topups',
        {
          headers: { 'X-API-Key': API_KEY },
          params: { page: 1, limit: 20 }
        }
      );

      data.topups.forEach(topup => {
        console.log(`  ${topup.created_at}: ${topup.amount_usd} (${topup.status})`);
    ```

## Status Codes

| Code | Description |
|------|-------------|
| `200` | Success |
| `400` | Invalid request (e.g., amount below minimum) |
| `401` | Missing or invalid API key |
| `402` | Insufficient wallet balance |
| `429` | Rate limit exceeded |
| `500` | Server error |

---

## Error Responses

**Insufficient Balance (402):**

```json
{
  "error": "insufficient_api_balance",
  "message": "Insufficient API wallet balance. Required: $0.50, Available: $0.25",
  "required_cents": 50,
  "available_cents": 25
}
```

**Invalid Amount (400):**

```json
{
  "error": "invalid_amount",
  "message": "Amount must be between $1.00 and $10,000.00"
}
```
